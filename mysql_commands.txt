CREATE TABLE booking_details (
booking_id int NOT NULL PRIMARY KEY AUTO_INCREMENT, 
movie_name VARCHAR(100) NOT NULL,
show_date_and_time DATETIME,
firstname VARCHAR(50) NOT NULL,
lastname VARCHAR(50),
confirmation_number VARCHAR(50),
status bool);

CREATE TABLE movies (
movie_id int NOT NULL PRIMARY KEY AUTO_INCREMENT, 
movie_name VARCHAR(100) NOT NULL);

show timings:
show_id
show time
show date
screen number

Create table show_timings (
show_id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
show_time TIME,
show_date date,
screen_no Varchar(10));

tickets_booked:
booking_id
movie_id 
show_id 
firstname 
lastname
confirmation number
status

Create table tickets_booked(
booking_id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
movie_id int,
FOREIGN KEY (movie_id) REFERENCES movies(movie_id),
show_id int,
FOREIGN KEY (show_id) REFERENCES show_timings(show_id),
firstname VARCHAR(50) NOT NULL,
lastname VARCHAR(50),
confirmation_no VARCHAR(50),
status bool);
